rule fastqc:
    input:  "{x}.R1.fastq",
            "{x}.R2.fastq"
    output: "QC/{x}.R1_fastqc.html",
            "QC/{x}.R2_fastqc.html"
    params: fastqc=config['bin']['FASTQC'],
            adapters=config['references']['fastqc.adapters']
    shell:  "{params.fastqc} -o QC -f fastq -c {params.adapters} {input}"
