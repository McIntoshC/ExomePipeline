rule gatk_HTC:
	input: bams='{samples}.BWA-aln.RG.MKDUP.realign.recal.bam'
	output: gvcf='{samples}.gvcf', idx='{samples}.gvcf.idx'
	params: batch="-l nodes=1:gpfs", GATK_BASE=config.GATK_BASE2, REF=config.REF, ontarget=config.onTarget, snp138=config.SNP138
        threads: THREADS
        run:
	    	bams=gatk_multi_arg("-I", input.bams)
	    	shell( '/usr/local/java/jdk1.7.0_02/bin/{params.GATK_BASE} -T HaplotypeCaller --read_filter BadCigar -variant_index_type LINEAR -variant_index_parameter 128000 --emitRefConfidence GVCF -R {params.REF} -L {params.ontarget} --dbsnp {params.snp138} -I {input.bams} -o {output.gvcf}')
