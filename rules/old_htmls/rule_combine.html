rule combine:
    input:  "xaa" 
    output: "xaa.gvcf"
    params: gatk=config['bin']['GATK'],genome=config['references']['GENOME']
    run:
      o=os.popen("ls xa?")
      for I in o.read().split():
          F=open(I,"r")
          fl=F.read().split()
          F.close()
          var=" --variant "+" --variant ".join(fl)
          out=I+".gvcf"
          cmd="{params.gatk} -R {params.genome} -T CombineGVCFs -o "+out+ var
          print(cmd)
          shell(cmd)
